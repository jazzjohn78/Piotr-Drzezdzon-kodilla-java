SELECT I.SUMMARY, L.NAME
FROM ISSUES I
JOIN ISSUESLISTS L ON I.ISSUESLISTS_ID = L.ID;

SELECT I.SUMMARY, U.FIRSTNAME AS FIRSTNAME_OF_USER_ASSIGNED_TO, U.LASTNAME AS LASTNAME_OF_THE_GUY
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_ASSIGNED
FROM USERS U
JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID
HAVING COUNT(*) > 0;
